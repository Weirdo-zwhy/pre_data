SELECT sys_describe_object(refclassid, refobjid, refobjsubid) AS "Collation", sys_describe_object(classid, objid, objsubid) AS "Object" FROM sys_depend d JOIN sys_collation c ON refclassid = 'sys_collation'::regclass AND refobjid = c.oid WHERE c.collversion <> sys_collation_actual_version(c.oid) ORDER BY 1, 2;
