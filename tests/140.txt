CREATE TABLE customers ( customer_id NUMBER(6), name VARCHAR2(20), regdate date, email VARCHAR2(30), territory VARCHAR2(20), userlevel NUMBER(2) ) PARTITION BY RANGE (regdate) INTERVAL ('3 month'::interval) SUBPARTITION BY HASH (customer_id) SUBPARTITION TEMPLATE ( SUBPARTITION hash1, SUBPARTITION hash2, SUBPARTITION hash3, SUBPARTITION hash4 ) ( PARTITION p0 VALUES LESS THAN(TO_DATE('2020-04-01', 'YYYY-MM-DD')) );
