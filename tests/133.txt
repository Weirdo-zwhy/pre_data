CREATE TABLE sales ( prod_id NUMBER(6) , cust_id NUMBER , time_id DATE , channel_id CHAR(1) , promo_id NUMBER(6) , state_id varchar2(2) , quantity_sold NUMBER(3) , amount_sold NUMBER(10,2) ) PARTITION BY LIST (state) ( PARTITION northwest VALUES ('OR', 'WA') , PARTITION southwest VALUES ('AZ', 'UT', 'NM') , PARTITION northeast VALUES ('NY', 'VM', 'NJ') , PARTITION southeast VALUES ('FL', 'GA') , PARTITION northcentral VALUES ('SD', 'WI') , PARTITION southcentral VALUES ('OK', 'TX') , PARTITION defaultstate VALUES (DEFAULT) );
